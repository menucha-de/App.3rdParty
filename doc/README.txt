To use remote publishing ensure you have installed and linked jsch ant extension

I: libjsch-java, libjsch-agent-proxy-java

 ln -s /usr/share/java/jsch.jar ~/.ant/lib/jsch.jar
 ln -s /usr/share/java/jsch.agentproxy.core.jar ~/.ant/lib/jsch.agentproxy.core.jar
 ln -s /usr/share/java/jsch.agentproxy.sshagent.jar ~/.ant/lib/jsch.agentproxy.sshagent.jar
